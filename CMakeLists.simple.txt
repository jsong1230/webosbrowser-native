cmake_minimum_required(VERSION 3.16)
project(webosbrowser-simple VERSION 1.0.0 LANGUAGES CXX)

# C++ 표준
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 간단한 실행 파일
add_executable(webosbrowser-simple
    src/main-simple.cpp
)

# 출력 디렉토리
set_target_properties(webosbrowser-simple PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
